19  package org.chabala.brick.controllab;
20  
21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;
23  
24  import java.io.Closeable;
25  import java.io.IOException;
26  import java.util.concurrent.*;
27  
28  /**
29   * Handles sending keep alive messages in the absence of other commands.
30   */
31  class KeepAliveMonitor implements Closeable {
32      private final Logger log = LoggerFactory.getLogger(getClass());
33      private final SerialPort serialPort;
34      private final ScheduledExecutorService executor;
35      private static final byte KEEP_ALIVE_COMMAND = 0x02;
36      /** Slightly less than two seconds. */
37      private static final int KEEP_ALIVE_PERIOD_MS = 1800;
38      private ScheduledFuture<?> task;
39  
40      /**
41       * Creates a keep alive monitor for the specified serial port. As soon
42       * as it's constructed, will send a keep alive message to the serial
43       * port with a frequency just under two seconds, as long as the port
44       * is open.
45       * @param serialPort the port to receive keep alive messages
46       */
47      KeepAliveMonitor(SerialPort serialPort) {
48          this.serialPort = serialPort;
49          executor = Executors.newSingleThreadScheduledExecutor(
50                  new NamedDaemonThreadFactory(
51                          "KeepAlive " + serialPort.getPortName()));
52          scheduleTask();
53      }
54  
55      /**
56       * Call this method when a command is being sent, to reset the countdown
57       * on the keep alive monitor. It will reschedule the next keep alive
58       * message from this moment.
59       */
60      void reset() {
61          task.cancel(false);
62          scheduleTask();
63      }
64  
65      private void scheduleTask() {
66          task = executor.scheduleAtFixedRate(() -> {
67              try {
68                  if (serialPort.isOpen()) {
69                      sendCommand(KEEP_ALIVE_COMMAND);
70                  }
71              } catch (IOException e) {
72                  LoggerFactory.getLogger(getClass()).error(e.getMessage(), e);
73              }
74          }, KEEP_ALIVE_PERIOD_MS, KEEP_ALIVE_PERIOD_MS, TimeUnit.MILLISECONDS);
75      }
76  
77      private void sendCommand(byte b) throws IOException {
78          log.info("TX -> {}", String.format("0x%02X", b));
79          serialPort.write(b);
80      }
81  
82      /**
83       * Cancel the next keep alive message and shutdown the monitor.
84       */
85      @Override
86      public void close() {
87          if (executor != null) {
88              executor.shutdownNow();
89          }
90      }
91  
92      /**
93       * A {@link ThreadFactory} which produces named, daemon threads.
94       */
95      private static class NamedDaemonThreadFactory implements ThreadFactory {
96          private ThreadFactory defaultThreadFactory = Executors.defaultThreadFactory();
97          private final String threadName;
98  
99          NamedDaemonThreadFactory(String threadName) {
100             this.threadName = threadName;
101         }
102 
103         @Override
104         public Thread newThread(Runnable r) {
105             Thread thread = defaultThreadFactory.newThread(r);
106             thread.setName(threadName);
107             thread.setDaemon(true);
108             return thread;
109         }
110     }
111 }
