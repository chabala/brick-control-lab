19  package org.chabala.brick.controllab;
20  
21  import jssc.SerialPortList;
22  
23  import java.util.Arrays;
24  import java.util.List;
25  
26  /**
27   * jSSC based implementation of {@link SerialPortFactory}.
28   *
29   * @see <a href="https://github.com/scream3r/java-simple-serial-connector">
30   *               https://github.com/scream3r/java-simple-serial-connector</a>
31   */
32  class JsscSerialPortFactory implements SerialPortFactory {
33  
34      /** {@inheritDoc} */
35      @Override
36      public List<String> getAvailablePorts() {
37          return Arrays.asList(SerialPortList.getPortNames());
38      }
39  
40      /** {@inheritDoc} */
41      @Override
42      public SerialPort getSerialPort(String portName) {
43          return new JsscSerialPort(portName);
44      }
45  }
